file(GLOB HEADERS "include/steemit/plugins/auth_util/*.hpp")

if(BUILD_SHARED_LIBRARIES)
    add_library(golos_auth_utility SHARED
            ${HEADERS}
            auth_util_plugin.cpp
            auth_util_api.cpp
            )
else()
    add_library(golos_auth_utility STATIC
            ${HEADERS}
            auth_util_plugin.cpp
            auth_util_api.cpp
            )
endif()

target_link_libraries(golos_auth_utility golos::application golos::chain golos::protocol fc)
target_include_directories(golos_auth_utility
        PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include")
